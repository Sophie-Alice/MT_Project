###################################### Feature Plots

plot_features <- function(seurat_obj, features, title = NULL, subtitle = NULL, num_columns = 2) {
  p <- FeaturePlot_scCustom(seurat_obj, features = features, num_columns = num_columns, pt.size = 0.1)

  p <- p + patchwork::plot_annotation(
    title = title,
    subtitle = subtitle,
    theme = ggplot2::theme(
      plot.title.position = "plot",
      plot.title    = ggplot2::element_text(face = "bold", size = 12, hjust = 0),
      plot.subtitle = ggplot2::element_text(face = "italic", size = 10, hjust = 0)
    )
  )
  
  return(p)
}
