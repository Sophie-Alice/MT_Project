Volcano_Plots <- function(Data, Title, Subtitle, pCutoff = 0.05, FCcutoff = 0.5) {
  
  # Clean extreme values
  df <- Data
  df <- df[is.finite(df$avg_log2FC) & is.finite(df$FDR), ]
  
  # dynamic axis limits
  xmax <- max(df$avg_log2FC, na.rm = TRUE)
  xmin <- min(df$avg_log2FC, na.rm = TRUE)
  ymax <- max(-log10(df$FDR), na.rm = TRUE)
  
  # Add padding
  xpad <- (xmax - xmin) * 0.1
  ypad <- ymax * 0.1
  
  combined_subtitle <- paste0(Subtitle, " ● Thresholds: FDR < ", pCutoff, ", |Log2FC| > ", FCcutoff)
  
  EV <- EnhancedVolcano(
    df,
    lab = df$gene,
    x = 'avg_log2FC',
    y = 'FDR',
    xlab = 'log2 Foldchange',
    ylab = "-log10 FDR",
    legendLabels = c('NS','log2FC','FDR','FDR & log2FC'),
    col = c("darkgrey", "darkgrey", "#FFD966", "#9F1C67"),
    title = Title,
    subtitle = combined_subtitle,   
    xlim = c(xmin - xpad, xmax + xpad),
    ylim = c(0, ymax + ypad),
    pCutoff = pCutoff,              
    FCcutoff = FCcutoff,            

    cutoffLineType = 'dashed',      
    cutoffLineCol = 'black',        
    cutoffLineWidth = 0.6,
    labSize = 2.0,
    drawConnectors = TRUE,
    widthConnectors = 0.5
  )
  
  return(EV)
}

raincloud.plots <- function(gene_list, 
                                   plot_title, 
                                   plot_subtitle, 
                                   seurat_obj = Data, 
                                   colors = gradient_colors4) {

  
  valid_genes <- intersect(gene_list, rownames(seurat_obj))
  expression_matrix <- FetchData(seurat_obj, vars = c(valid_genes, "ident")) 

  
  rain_data <- expression_matrix %>%
    tibble::rownames_to_column("Cell_ID") %>%
    tidyr::pivot_longer(
      cols = all_of(valid_genes), 
      names_to = "Gene", 
      values_to = "Expression"
    ) %>%
    dplyr::rename(Condition = ident)
  

  raincloud_plot <- ggplot(rain_data, aes(x = Condition, y = Expression, fill = Condition, color = Condition)) +
    geom_rain(
      alpha = 0.6,
      violin.args = list(color = NA),          
      boxplot.args = list(outlier.shape = NA), 
      point.args = list(
        size = 0.4,                            
        alpha = 0.4                            
      ),
    group = 1
    ) +
    
    facet_wrap(~Gene, scales = "free_y", ncol = 3) +
    
    scale_fill_manual(values = colors) +
    scale_color_manual(values = colors) +
    
    theme_bw() +
    labs(
      title    = plot_title,
      subtitle = plot_subtitle,
      x        = "Experimental Conditions",
      y        = "Normalized Expression Level"
    ) +
    
    theme(
      plot.title.position = "plot",
      plot.title    = element_text(face = "bold", size = 12, hjust = 0),
      plot.subtitle = element_text(face = "italic", size = 10, hjust = 0),
      axis.text.x   = element_text(angle = 45, hjust = 1, face = "bold", color = "black"),
      axis.text.y   = element_text(face = "bold", color = "black"),
      axis.title.x  = element_text(face = "bold"),
      axis.title.y  = element_text(face = "bold"),
      strip.text    = element_text(face = "bold", size = 10), 
      legend.position = "none"
    )
  
  return(raincloud_plot)
}


###################################### 2nd Volcano Viz


generate_volcano <- function(df, title_text) {
  df <- df %>%
    mutate(Significant = ifelse(FDR < 0.05 & abs(avg_log2FC) > 0.5,
                                ifelse(avg_log2FC > 0, "Up", "Down"), "ns"))
  
  top_hits <- df %>%
    filter(Significant != "ns") %>%
    arrange(desc(abs(avg_log2FC))) %>%
    slice_head(n = 30)
  
  ggplot(df, aes(x = avg_log2FC, y = -log10(FDR))) +
    geom_point(aes(color = Significant), alpha = 0.5, size = 1.2) +
    scale_color_manual(values = c("Up" = "#de2d26", "Down" = "#3182bd", "ns" = "#969696")) +
    geom_hline(yintercept = -log10(0.05), linetype = "dashed", color = "black") +
    geom_vline(xintercept = c(-0.25, 0.25), linetype = "dashed", color = "black") +
    geom_text_repel(data = top_hits, aes(label = gene), size = 2.5, max.overlaps = 15) +
    theme_classic() +
    theme(legend.position = "none") +
    labs(title = title_text, x = "log2 Fold Change", y = "-log10(FDR)")
}
